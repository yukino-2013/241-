四、各模块详细说明（供AI生成代码参考）
模块1：tools.py（后端工具函数）
功能：提供三个独立的系统信息查询函数。
python
# tools.py 完整代码模板
import psutil
import subprocess
def get_cpu_usage() -> str:
    """
    获取CPU使用率
    返回格式: "当前CPU使用率为{百分比}%"
    """
    usage = psutil.cpu_percent(interval=1)
    return f"当前CPU使用率为{usage}%"
def get_memory_usage() -> str:
    """
    获取内存使用情况
    返回格式: "内存已使用{used_gb}GB/共{total_gb}GB，使用率{percent}%"
    """
    mem = psutil.virtual_memory()
    used_gb = mem.used / (1024**3)
    total_gb = mem.total / (1024**3)
    percent = mem.percent
    return f"内存已使用{used_gb:.1f}GB/共{total_gb:.1f}GB，使用率{percent}%"
def check_service_status(service_name: str) -> str:
    """
    检查Linux系统服务状态
    参数: service_name - 服务名称，如 nginx, sshd, mysql
    返回格式: "服务{service_name}正在运行" 或 "服务{service_name}未运行"
    """
    try:
        result = subprocess.run(
            ["systemctl", "is-active", service_name],
            capture_output=True, text=True, timeout=5
        )
        if result.returncode == 0 and result.stdout.strip() == "active":
            return f"服务{service_name}正在运行"
        else:
            return f"服务{service_name}未运行"
    except Exception as e:
        return f"检查服务{service_name}时出错: {str(e)}"
# 注意：如果运行环境不是Linux，可以使用以下备用函数
def check_service_by_process_name(service_name: str) -> str:
    """
    基于进程名检查服务（适用于Windows/Mac）
    """
    for proc in psutil.process_iter(['name']):
        if service_name.lower() in proc.info['name'].lower():
            return f"服务{service_name}正在运行"
    return f"服务{service_name}未运行"
模块2：agent.py（智能体封装）
功能：将工具函数包装成LangChain工具，创建智能体。
python
# agent.py 完整代码模板
import os
from dotenv import load_dotenv
from langchain_openai import ChatOpenAI
from langchain.tools import Tool
from langchain.agents import initialize_agent, AgentType
from tools import get_cpu_usage, get_memory_usage, check_service_status
# 加载环境变量
load_dotenv()
# 初始化大模型（使用智谱AI，OpenAI兼容接口）
llm = ChatOpenAI(
    model="glm-4-flash",  # 智谱免费模型
    openai_api_key=os.getenv("ZHIPU_API_KEY"),
    openai_api_base="https://open.bigmodel.cn/api/paas/v4/",
    temperature=0,  # 0表示确定性输出，适合工具调用
)
# 定义工具列表
tools = [
    Tool(
        name="get_cpu_usage",
        func=get_cpu_usage,
        description="获取当前CPU使用率。当用户询问CPU使用情况、电脑卡不卡、系统负载时调用此工具。"
    ),
    Tool(
        name="get_memory_usage",
        func=get_memory_usage,
        description="获取当前内存使用情况。当用户询问内存够不够、内存使用率、剩余内存时调用此工具。"
    ),
    Tool(
        name="check_service_status",
        func=check_service_status,
        description="检查某个系统服务是否在运行。输入服务名称（如nginx、sshd、mysql）。当用户询问某个服务活着吗、运行状态时调用此工具。"
    )
]
# 创建智能体
agent = initialize_agent(
    tools,
    llm,
    agent=AgentType.ZERO_SHOT_REACT_DESCRIPTION,
    verbose=True,  # 打印思考过程，便于调试
    max_iterations=3,  # 最多迭代3次，避免无限循环
)
def ask_agent(question: str) -> str:
    """
    对外接口：传入用户问题，返回智能体答案
    """
    try:
        response = agent.run(question)
        return response
    except Exception as e:
        return f"处理问题时出错: {str(e)}"
# 测试入口（可选）
if __name__ == "__main__":
    while True:
        q = input("请输入问题（输入exit退出）: ")
        if q.lower() == "exit":
            break
        print(ask_agent(q))

# main.py 完整代码模板
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from agent import ask_agent
app = FastAPI(title="智能运维助手API", version="1.0.0")
# 允许跨域（前端Streamlit需要）
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # 开发环境允许所有来源
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)
class Question(BaseModel):
    text: str
@app.post("/chat")
async def chat(question: Question):
    """
    接收用户问题，返回智能体答案
    """
    answer = ask_agent(question.text)
    return {"answer": answer}
@app.get("/health")
async def health():
    """
    健康检查接口
    """
    return {"status": "ok"}
if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)

# app.py 完整代码模板
import streamlit as st
import requests
# 后端API地址（根据实际部署修改）
API_URL = "http://localhost:8000/chat"
# 页面配置
st.set_page_config(page_title="智能运维助手", page_icon="🤖", layout="centered")
# 标题
st.title("🤖 智能运维助手")
st.markdown("用自然语言查询服务器状态，例如：**CPU怎么样？**")
# 初始化聊天历史
if "messages" not in st.session_state:
    st.session_state.messages = []
# 显示历史消息
for msg in st.session_state.messages:
    with st.chat_message(msg["role"]):
        st.markdown(msg["content"])
# 输入框
if prompt := st.chat_input("输入你的问题..."):
    # 显示用户消息
    with st.chat_message("user"):
        st.markdown(prompt)
    st.session_state.messages.append({"role": "user", "content": prompt})
    # 调用后端API
    with st.spinner("智能体思考中..."):
        try:
            response = requests.post(API_URL, json={"text": prompt}, timeout=15)
            if response.status_code == 200:
                answer = response.json()["answer"]
            else:
                answer = f"❌ 后端错误：{response.status_code}"
        except requests.exceptions.ConnectionError:
            answer = "❌ 无法连接到后端，请确认后端服务已启动（运行 python main.py）"
        except requests.exceptions.Timeout:
            answer = "⏰ 请求超时，请稍后重试"
        except Exception as e:
            answer = f"❌ 发生错误：{str(e)}"
    # 显示助手回复
    with st.chat_message("assistant"):
        st.markdown(answer)
    st.session_state.messages.append({"role": "assistant", "content": answer})
