### 3. 使用Python脚本测试
创建 `test_api.py`：
```python
import requests
import json

BASE_URL = "http://127.0.0.1:8000"

def test_health():
    """测试健康检查"""
    response = requests.get(f"{BASE_URL}/health")
    print("健康检查:", response.json())

def test_chat(question):
    """测试聊天接口"""
    response = requests.post(
        f"{BASE_URL}/chat",
        json={"text": question}
    )
    result = response.json()
    print(f"问题: {question}")
    print(f"回答: {result['answer']}")

def test_tools():
    """测试工具状态"""
    response = requests.get(f"{BASE_URL}/tools/status")
    tools = response.json()["tools"]
    print(f"可用工具: {len(tools)} 个")
    for tool in tools:
        print(f"  - {tool['name']}")

if __name__ == "__main__":
    test_health()
    test_tools()
    test_chat("CPU使用率怎么样？")
    test_chat("内存够用吗？")
    test_chat("检查python服务状态")
```
